}

endpoint first_User : GET "/users/{x}/{y}" {
   response : relational {
        r_1 = Select<name eq y,age lst x> (User);
        r_2 = Project<name,lastname>(r_1);
        r_3 = Select<>(Person);
        r_4 = Join_inner<name,username>(r_2,r_4);
        r_5 = Project<name,lastname,bth>(r_4);
        len_temp = Len(r_5) ;
        r_6 = Limit<1,len_temp,3>(r_5);
        -> r_6 ;
   };
}
