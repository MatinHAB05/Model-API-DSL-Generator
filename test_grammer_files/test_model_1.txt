enum Role{
    "ADMIN" ,
    "USER" ,
    "GUEST" ,
    "MANAGER"
}

model Person {
    username : String @pk @non-nullable @unique @vaild[wildpattern="..."];
    age : Int @nullable @vaild[min=8,max=14];
    role : Role @non-nullable @vaild[exclude={"ADMIN"}];
    second_role : Role @non-nullable @vaild[include={"MANAGER","USER"}];
    bth : Date @vaild[min="2020-01-01" , max="2024-06-11"];
}
